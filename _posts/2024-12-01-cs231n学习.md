```matlab 
%% QPSK
 clc;
 clear
 close all;
 twpi=2*pi;
 N = 256; % 码元数
 Tb=0.01 ;% 码元周期
 fb=1/Tb; % 码元频率
 fs=10*fb; % 采样频率
 ts=1/fs;%0.001;% 采样时刻
 nt=Tb/ts; % 一个码元周期内 采样点数
 Total_T = Tb*N;
 t=0:ts:Tb*N-ts; %时间长度
 fc= 2*fb;% 载波频率
 rng('shuffle');
```
